Hide files in ls, by adding support for .hidden files!

alias ls='if [[ -f .hidden ]]; then while read l; do opts+=(--hide="$l"); done < .hidden; fi; ls --color=auto "${opts[@]}"' 
ant@dev(/ntfs)# \ls -F

ant@dev(/ntfs)# cat .hidden
$RECYCLE.BIN
System Volume Information

ant@dev(/ntfs)# ls -F
Shared/

Extract a bash function

sed -n '/^function h\(\)/,/^}/p' script.sh 
$ sed -n '/^function h\(\)/,/^}/p' .bash_profile
function h() {
    if [ -z "$1" ]
    then
        history
    else
        history | grep "$@"
    fi
}

Indent a one-liner.

declare -f <function name> 
rany:~/Desktop$ command_8239() { M=50;if read mb _ < <(du -sm 2>/dev/null) && (( mb > M )); then echo "Size of ${PWD##*/} > $M";fi; }
rany:~/Desktop$ declare -f command_8239
command_8239()
{
    M=50;
    if read mb _ < <(du -sm 2>/dev/null) && (( mb > M )); then
        echo "Size of ${PWD##*/} > $M";
    fi
}
rany:~/Desktop$

I often write a one-liner which I want to use later in a script.

Alternatives

type <function name> 
$ command1 () { for m in 2010-{01..12} ; do m=$( echo $m | sed 's/-\([0-9]\)$/-0\1/' ) ; echo $m ; done }
$ type command1
command1 is a function
command1 ()
{
    for m in 2010-{01..12};
    do
        m=$( echo $m | sed 's/-\([0-9]\)$/-0\1/' );
        echo $m;
    done
}

Bash builtin type also indents the function.
